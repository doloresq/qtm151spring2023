# Write your own code
list_csv = ["data_quiz/wdi_co2_2005.csv", "data_quiz/wdi_co2_2010.csv", 
    "data_quiz/wdi_co2_2015.csv"]

index = 0
for csv in list_csv:
    dataset = pd.read_csv(str(csv))
    co2_gasfuel_kt = dataset["co2_gasfuel_kt"]
    co2_liquidfuel_kt = dataset["co2_liquidfuel_kt"]
    co2_emissions_total= co2_gasfuel_kt + co2_liquidfuel_kt
    plt.hist(co2_emissions_total)
    plt.title("Total Emissions of CO2 Graph" + str(index+1))
    plt.ylabel("Frequency of Amount of CO2")
    plt.xlabel("Gasfuel & Liquidfuel CO2")
    plt.show()
    index = index + 1
